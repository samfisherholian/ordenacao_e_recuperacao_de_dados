A indeia é dividir o array em 2 sub-arrays inicialmente o indice 0
vai fazer parte do array ordenado e os demais vão fazer parte do array não ordenado e então vamos fazer comparações com
os valores não ordenado verica se o valor do array ordenado for maior que o valor não ordenando 
então troca as posições e colocamo esse dois valores comparados na 
parte do array ordenado e fazemos isso para as demais posições

insertion(inteiro A[], inteiro tamanho){

    para i <- 1 ate tamanho-1{

        inteiro valor_na_posicao <- A[i]

        inteiro marcao_do_indice_do_array <- i

        // enquanto o valor do array na posição anterior for maior que o valor da proxima posição
        // o array na posição atual recebe o valor da posição anterior 
        enquanto( marcao_do_indice_do_array > 0 E A[marcao_do_indice_do_array-1] > valor_na_posicao) faca {

            A[marcao_do_indice_do_array] <- A[marcao_do_indice_do_array - 1]

            marcao_do_indice_do_array <- marcao_do_indice_do_array - 1
        }
        // valor na posição 0 recebe o valor da posição 1
        A[marcao_do_indice_do_array] <- valor_na_posicao

    }

}
